const axios = require("axios");

async function getInstagramData(username) {
  try {
    const profile = await axios.post(
      "https://api.boostfluence.com/api/instagram-profile-v2",
      { username },
      { headers: { "Content-Type": "application/json" } }
    );

    console.log({
      username: profile.data.username,
      fullname: profile.data.full_name,
      followers: profile.data.follower_count,
      following: profile.data.following_count,
      posts: profile.data.media_count,
      bio: profile.data.biography,
      pic: profile.data.profile_pic_url_hd
    });

  } catch (e) {
    console.error("Error:", e.response ? e.response.data : e.message);
  }
}

getInstagramData("b4gus.real");
